% Commands for drawing the scale in the 'blocks' diagrams.
\newdrawing{\scale}{%

   \draw (0,0) -- (30,0);

   \foreach \x in {0,5,...,25}
   {
      \draw (\x,-0.6) -- (\x,0.6);

      \foreach \d in {1,2,3,4}
      {
         \draw (\x + \d, -0.3) -- (\x + \d, 0.3); 
      }
   }

   \draw (30,-0.6) -- (30,0.6);
}


% Command for creating a block
\newdrawing{\block}{%
   \fill[black] (-0.2, -0.2) rectangle (0.2,0.2);
}
