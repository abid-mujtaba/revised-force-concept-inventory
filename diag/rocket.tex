    ;

    \begin{scope}[decoration={markings, mark=at position 0.5 with {\arrow{>}}}]     % All lines in this scope have an arrow in the middle

            (a) node[anchor=north, yshift=-5pt] {$a$} -- (b) node[anchor=north, yshift=-5pt] {$b$}
        ;

    \end{scope}

    \begin{scope}[xshift=110,yshift=20,yscale=-1,scale=0.3,xscale=0.8]

        \fill[black] (0.4,4) rectangle (-0.4,4.25);

        \draw (-0.4,4.25) -- (-1,4.75) -- (1,4.75) -- (0.4,4.25);
